     text=white, 
     border/.style={line width=14mm}]

    {60/blue, 
     120/cyan, 
     180/green!30!black,

     {
        \draw[\col, border] (\last:2cm) 
             arc[start angle=\last, end angle=\angle, radius=2cm];
        \draw[white, line width=1mm] (\last:1.3)--++(\last:1.4);
}
