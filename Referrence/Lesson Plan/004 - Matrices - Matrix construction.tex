\textbf{MATRICES OF DIFFERENT ORDERS} \\ \\

\textbf{MATRIX OF ORDER 2 $\times$2} \\ \\
ROUND BRACKET ----- \ 
\( A = \left( 
\begin{array}{cc} 
a_{11} & a_{12}\\ 
a_{21} & a_{22} 
\end{array} 
\right) \) \\ \\ \\
SQUAREBRACKET------
\( A = \left[ 
\begin{array}{cc} 
a_{11} & a_{12}\\ 
a_{21} & a_{22} 
\end{array} 
\right] \) \\ \\

\textbf{MATRIX OF ORDER 2 $\times$3} \\ \\

\end{array}

SQUAREBRACKET------
\( A = \left[

\end{array} 
\right] \) \\ \\
