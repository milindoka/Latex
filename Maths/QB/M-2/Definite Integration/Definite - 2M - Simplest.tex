\section{Question} 

$\begin{aligned}[t] 
\text{Evaluate \ } \int_{0}^{a} \frac{1}{\sqrt{a^2-x^2}}dx
\end{aligned}$

%----------------------------------------
\subsection*{Answer}
\begin{equation} \nonumber
\begin{alignedat}{4}
& \int_{0}^{a} \frac{1}{\sqrt{a^2-x^2}}dx \\
&= \left[ sin^{-1}\frac{x}{a}\right]_{0}^{a}\\
\end{alignedat}
\quad \quad \quad \quad
\vrule
\quad
\begin{alignedat}{4}
&= \left[ sin^{-1}1\right] - \left[ sin^{-1}0\right]\\
&= \frac{\pi}{2}\\
&  \\
\end{alignedat}
\end{equation}

\section{Question} 

$\begin{aligned}[t] 
\text{Evaluate \ } \int_{0}^{\frac{\pi}{2}} \frac{\cos x}{1+\sin x}\ dx
\end{aligned}$

%----------------------------------------
\subsection*{Answer}
\begin{equation} \nonumber
\begin{alignedat}{4}
& \int_{0}^{\frac{\pi}{2}} \frac{\cos x}{1 + \sin x}\,dx \\
&= \left[\  \log (1 + \sin x)\right]_{0}^{\frac{\pi}{2}}\\
\end{alignedat}
\quad \quad \quad \quad
\vrule
\quad
\begin{alignedat}{4}
&= \log 2 - \log 1\\
&= \log 2\\
&  \\
\end{alignedat}
\end{equation}
