\newcommand{\dt}[2]{\draw ($(#1)!0.5!(#2)$) circle(0.08);\draw (#1)--(#2);}

%Draw arrow at midpoint by combining two lines
\newcommand{\ma}[2]
{ \draw [shorten >= -2.5mm][-{Stealth[length=5mm]}] (#1) -- ($(#1)!0.5!     			(#2)$);  
  \draw ($(#1)!0.5!(#2)$)--(#2);  
}
